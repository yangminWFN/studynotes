# [4. 寻找两个正序数组的中位数](https://leetcode-cn.com/problems/median-of-two-sorted-arrays/)

```java
class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        if(nums1 != null && nums2 != null){
            int len1 = nums1.length;
            int len2 = nums2.length;
            //int[] nums = new int[len1 + len2];
            int first = 0;
            int second = 0;
            int i = 0;
            int j = 0;
            //int mid = nums.length / 2;
            int len = len1 + len2;
            int mid = len / 2;
            for(int k = 0; k <= mid; k++){
                if( k != 0){
                    first = second;
                }
                if(i == len1){
                    second = nums2[j];
                    j++;
                }else if(j == len2){
                    second = nums1[i];
                    i++;
                }else if(nums1[i] <= nums2[j]){
                    second = nums1[i];
                    i++;
                }else{
                    second = nums2[j];
                    j++;
                }
                if( k == 0){
                    first = second;
                }
            }
            return len % 2 == 0 ? (second + first) / 2.0 : second; 
        }else{
            if((nums1 == null || nums1.length == 0) && (nums2 == null || nums2.length == 0)){
                return 0;
            }else if(nums1 == null || nums1.length == 0){
                return getMid(nums2);
            }else{
                return getMid(nums1);
            }
        }
        
    }

    public double getMid(int[] a){
        int mid = a.length / 2;
        if(a.length % 2 == 0){
            return (a[mid] + a[mid - 1]) / 2.0;
        }else{
            return a[mid];
        }
    }
}
```
